\begin{exercice}\label{exoSerieUn0009}

	Montrer que si $a$ et $b$ sont des nombres réels, alors 
	\begin{subequations}
		\begin{align}
			\sup\{ a,b \}=\frac{ 1 }{2}\big( a+b+| a-b | \big)\\			\label{SubEqsupexoSerUnab}
			\inf\{ a,b \}=\frac{ 1 }{2}\big( a+b-| a-b | \big).
		\end{align}
	\end{subequations}

\end{exercice}
