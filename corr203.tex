\begin{corrige}{203}
\begin{enumerate}
\item $\forall M\in\eR,\exists x_0\in\eR\tq (x>x_0)\Rightarrow f(x)\leq M.$
\item $\forall M\in\eR,\exists x_0\in\eR\tq (x<x_0)\Rightarrow f(x)\geq M.$
\end{enumerate}
\end{corrige}
